CREATE TABLE IF NOT EXISTS rooms (
    id INTEGER PRIMARY KEY,
    roomName TEXT NOT NULL UNIQUE,
    cost INTEGER
);

CREATE TABLE IF NOT EXISTS guests (
    id INTEGER PRIMARY KEY,
    roomID INTEGER,
    arrival DATE,
    departure DATE,
    extras TEXT,
    FOREIGN KEY (roomID) REFERENCES rooms (id)
);
